void SM_System_print(int* unistring)
{
	int length = unistring[-1];
	int i;
	char c;
	for (i = 0; i < length; ++i)
	{
		if (unistring[i] < 128)
		{
			printf("%c", (char) unistring[i]);
		}
		else
		{
			printf("?");
		}
	}
}

void SM_System_println(int* unistring)
{
	SM_System_print(unistring);
	printf("\n");
}
